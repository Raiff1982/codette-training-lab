def compute_semantic_tension(responses, embed_fn):
    embeddings = [embed_fn(r["content"]) for r in responses]

    tensions = []
    for i in range(len(embeddings)):
        for j in range(i + 1, len(embeddings)):
            sim = cosine_similarity(embeddings[i], embeddings[j])
            tension = 1 - sim
            tensions.append(tension)

    return sum(tensions) / len(tensions) if tensions else 0.0

def compute_confidence(logprobs):
    if not logprobs:
        return 0.5
    avg = sum(logprobs) / len(logprobs)
    return math.exp(avg)

def score_agents(responses, memory):
    scored = []

    for r in responses:
        mem_w = memory_weight(r["perspective"], memory)

        score = (
            0.5 * r["confidence"] +
            0.3 * mem_w +
            0.2 * r.get("ethical_score", 0.5)
        )

        r["score"] = score
        scored.append(r)

    return sorted(scored, key=lambda x: x["score"], reverse=True)

def debate_round(responses, critique_fn):
    updated = []

    for r in responses:
        critiques = []

        for other in responses:
            if other == r:
                continue

            critique = critique_fn(
                source=r["content"],
                target=other["content"]
            )
            critiques.append(critique)

        revised = apply_critiques(r["content"], critiques)

        updated.append({
            **r,
            "content": revised
        })

    return updated

def run_debate(responses, rounds, critique_fn):
