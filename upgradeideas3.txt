conflict_engine.py
class ConflictEngine:

    def __init__(self, epistemic, aegis, resonance):
        self.epistemic = epistemic
        self.aegis = aegis
        self.resonance = resonance

    def resolve(self, query, analyses):

        # ----------------------------------------
        # 1. Measure epistemic structure
        # ----------------------------------------
        tensions = self.epistemic.score_pairwise_tension(analyses)
        coherence = self.epistemic.score_ensemble_coherence(analyses)

        avg_tension = sum(tensions.values()) / max(len(tensions), 1)

        # ----------------------------------------
        # 2. Ethical evaluation (η)
        # ----------------------------------------
        ethical_scores = {}
        for agent, text in analyses.items():
            verdicts = self.aegis.evaluate(text)
            eta = sum(v.score for v in verdicts) / len(verdicts)
            ethical_scores[agent] = eta

        # ----------------------------------------
        # 3. Resonance state (ψr)
        # ----------------------------------------
        psi_state = self.resonance.compute_psi(
            coherence=coherence,
            tension=avg_tension
        )

        # ----------------------------------------
        # 4. Weight agents
        # ----------------------------------------
        weights = {}

        for agent in analyses:
            eta = ethical_scores.get(agent, 0.5)

            # Core weighting function
            weight = (
                0.5 * eta +
                0.3 * coherence +
                0.2 * (1 - avg_tension)
            )

            weights[agent] = max(0.0, min(1.0, weight))

        # Normalize
        total = sum(weights.values()) or 1.0
        weights = {k: v / total for k, v in weights.items()}

        # ----------------------------------------
        # 5. Select dominant + dissent
        # ----------------------------------------
        sorted_agents = sorted(weights.items(), key=lambda x: x[1], reverse=True)

        dominant = sorted_agents[:2]
        dissent = sorted_agents[-2:]

        return {
            "weights": weights,
            "dominant": dominant,
            "dissent": dissent,
            "coherence": coherence,
            "tension": avg_tension,
            "psi_r": psi_state.psi_r,
            "stability": psi_state.stability
        }
