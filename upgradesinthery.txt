💥 Step 1 — Memory-weighted trust engine
memory_weighting.py
class MemoryWeighting:

    def __init__(self, memory):
        self.memory = memory

    def compute_weights(self, agents, context):

        weights = {}

        for agent in agents:

            history = self.memory.retrieve_by_adapter(agent, limit=50)

            if not history:
                weights[agent] = 1.0
                continue

            # Aggregate signals
            coherence = sum(h.get("coherence", 0.5) for h in history) / len(history)
            tension = sum(h.get("tension", 0.5) for h in history) / len(history)

            # Ideal tension ~ 0.3–0.5 (productive disagreement)
            tension_score = 1.0 - abs(tension - 0.4)

            # Final trust score
            score = (
                0.6 * coherence +
                0.4 * tension_score
            )

            weights[agent] = max(0.1, min(score, 1.5))

        # Normalize
        total = sum(weights.values()) or 1.0
        return {k: v / total for k, v in weights.items()}
🧠 What this does

Each agent gets stronger if:

it produces coherent answers

it contributes productive tension (not chaos)

def apply_context_boost(self, weights, intent):

    boosted = weights.copy()

    if intent.get("type") == "technical":
        boosted["newton"] *= 1.2
        boosted["systems"] *= 1.2

    if intent.get("type") == "emotional":
        boosted["empathy"] *= 1.3

    if intent.get("uncertainty") == "high":
        boosted["quantum"] *= 1.2

    # normalize again
    total = sum(boosted.values()) or 1.0
    return {k: v / total for k, v in boosted.items()}
