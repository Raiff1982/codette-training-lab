class NexusConflictBridge:

    def __init__(self, nexus, conflict_engine, memory, optimizer):
        self.nexus = nexus
        self.conflict = conflict_engine
        self.memory = memory
        self.optimizer = optimizer

    def process(self, query, analyses):

        # ----------------------------------------
        # 1. Analyze incoming signal (NEXUS)
        # ----------------------------------------
        signal = self.nexus.analyze(query)

        intent = signal["intent"]
        risk = signal["intervention"]

        # ----------------------------------------
        # 2. Adjust reasoning pressure
        # ----------------------------------------
        tension_bias = 0.0

        if intent["pre_corruption_risk"] == "high":
            tension_bias += 0.2  # force more disagreement
        if intent["ethical_alignment"] == "unaligned":
            tension_bias += 0.2  # increase scrutiny

        # ----------------------------------------
        # 3. Resolve conflict
        # ----------------------------------------
        conflict_result = self.conflict.resolve(query, analyses)

        coherence = conflict_result["coherence"]
        tension = conflict_result["tension"] + tension_bias

        # ----------------------------------------
        # 4. Store memory (experience encoding)
        # ----------------------------------------
        dominant = [a for a, _ in conflict_result["dominant"]]

        self.memory.store_from_turn(
            query=query,
            response=str(conflict_result["weights"]),
            adapter=",".join(dominant),
            coherence=coherence,
            tension=tension
        )

        # ----------------------------------------
        # 5. Feed optimizer
        # ----------------------------------------
        self.optimizer.record_signal(
            QualitySignal(
                timestamp=time.time(),
                adapter=dominant[0] if dominant else "unknown",
                coherence=coherence,
                tension=tension,
                productivity=1.0 - abs(tension - 0.4),
                response_length=len(str(analyses)),
                multi_perspective=True
            )
        )

        return {
            "conflict": conflict_result,
            "intent": intent,
            "adjusted_tension": tension,
            "risk": risk
        }
