class DebateEngine:

    def __init__(self, generator):
        """
        generator = your model generate() function
        """
        self.generate = generator

    def run_debate(self, base_messages, responses, rounds=1):
        """
        responses = {adapter: text}
        """

        debate_log = {}

        for round_idx in range(rounds):
            new_responses = {}

            for agent, original in responses.items():

                others = {
                    k: v for k, v in responses.items() if k != agent
                }

                critique_prompt = self._build_prompt(agent, original, others)

                messages = base_messages + [
                    {"role": "system", "content": critique_prompt},
                    {"role": "user", "content": "Provide critique and refinement."}
                ]

                critique, _, _ = self.generate(messages, agent)

                new_responses[agent] = critique

            responses = new_responses
            debate_log[round_idx] = responses

        return responses, debate_log

    def _build_prompt(self, agent, self_text, others):

        other_text = "\n\n".join(
            f"{k.upper()}:\n{v}" for k, v in others.items()
        )

        return f"""
You are {agent.upper()} perspective.

Your original answer:
{self_text}

Other perspectives:
{other_text}

Instructions:
1. Identify where others are wrong or incomplete
2. Defend your reasoning
3. Update your answer if needed
4. Be critical, not agreeable
"""

import numpy as np


class ConfidenceEngine:

    def __init__(self, llm):
        self.llm = llm  # llama.cpp model

    def score(self, prompt, response):

        # Re-evaluate response token probabilities
        result = self.llm.create_completion(
            prompt=prompt + response,
            max_tokens=0,
            logprobs=1,
            echo=True
        )

        tokens = result["choices"][0]["logprobs"]["token_logprobs"]

        # Remove None values
        tokens = [t for t in tokens if t is not None]

        if not tokens:
            return 0.5

        avg_logprob = np.mean(tokens)

        # Convert logprob → confidence (0–1)
        confidence = np.exp(avg_logprob)

        return float(np.clip(confidence, 0.0, 1.0))
